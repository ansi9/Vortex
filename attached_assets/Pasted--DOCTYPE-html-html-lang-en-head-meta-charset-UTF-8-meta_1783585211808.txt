    <script>
        document.addEventListener('DOMContentLoaded', () => {
            try { if(window.lucide) lucide.createIcons(); } catch(e) { console.log("Icons pending..."); }

            // Enhanced Slider Logic
            const slider = document.getElementById('unlock-slider');
            const progress = document.getElementById('runway-progress');
            
            if(slider) {
                slider.addEventListener('input', function() {
                    // Update glowing trail width
                    if(progress) progress.style.width = this.value + '%';
                    
                    if(parseInt(this.value) > 85) {
                        forceUnlock();
                    }
                });

                // Snap back to zero if released before end
                slider.addEventListener('change', function() {
                    if(parseInt(this.value) <= 85) {
                        this.value = 0;
                        if(progress) progress.style.width = '0%';
                    }
                });
            }
        });

        function forceUnlock() {
            const landing = document.getElementById('landing-page');
            const app = document.getElementById('app');
            
            if(landing && app) {
                landing.style.transform = 'translateY(-100%)';
                app.style.opacity = '1';
                app.style.transform = 'scale(1)';
                
                setTimeout(() => {
                    const notif = document.getElementById('ai-notif');
                    if(notif) notif.style.display = 'block';
                }, 2000);
            }
        }

        function nav(pageId, btn) {
            document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
            const target = document.getElementById(pageId);
            if(target) target.classList.add('active');
            
            document.querySelectorAll('.nav-item').forEach(b => b.classList.remove('active'));
            if(btn) btn.classList.add('active');
        }

        function toggleAssistant() {
            const el = document.getElementById('ai-assistant');
            el.classList.toggle('open');
            const notif = document.getElementById('ai-notif');
            if(notif) notif.style.display = 'none';
        }

        function tripSelect(type) {
            const chat = document.getElementById('chat-body');
            
            const userMsg = document.createElement('div');
            userMsg.className = 'msg user';
            userMsg.innerText = type === 'short' ? "Short Trip (< 30m)" : "Long Trip (> 1hr)";
            chat.appendChild(userMsg);

            setTimeout(() => {
                const botMsg = document.createElement('div');
                botMsg.className = 'msg bot';
                botMsg.innerHTML = `I've prepared a safety checklist for your <b>${type}</b> trip.`;
                chat.appendChild(botMsg);

                const list = document.createElement('div');
                list.className = 'checklist';
                
                let items = type === 'short' 
                    ? ['Check Wallet/Pass', 'Wear Comfortable Shoes', 'Check Return Timings']
                    : ['Share Live Location', 'Pack Power Bank', 'Download Offline Map', 'Carry Water Bottle', 'Identify Safe Zones'];
                
                items.forEach(item => {
                    const row = document.createElement('div');
                    row.className = 'check-item';
                    row.onclick = function() {
                        const box = this.querySelector('input');
                        box.checked = !box.checked;
                        this.classList.toggle('done');
                    };
                    row.innerHTML = `<input type="checkbox"> <span>${item}</span>`;
                    list.appendChild(row);
                });
                chat.appendChild(list);
                
                const tip = document.createElement('div');
                tip.className = 'msg bot';
                tip.style.marginTop = '10px';
                tip.innerHTML = `<b>🛡️ Safety Tip:</b> ${type === 'short' ? 'Keep your bag zoomed in front of you in crowds.' : 'Set an alarm to check in with friends every hour.'}`;
                chat.appendChild(tip);

                chat.scrollTop = chat.scrollHeight;
            }, 600);
        }

        function toggleSOS(state) {
            const overlay = document.getElementById('sos-overlay');
            if(state) {
                overlay.style.display = 'flex';
                setTimeout(() => overlay.classList.add('active'), 10);
                
                let dots = 0;
                window.sosInterval = setInterval(() => {
                    dots = (dots + 1) % 4;
                    const status = document.getElementById('loc-status');
                    if(status) status.innerText = 'Broadcasting Live Location' + '.'.repeat(dots);
                }, 500);
            } else {
                overlay.classList.remove('active');
                setTimeout(() => overlay.style.display = 'none', 300);
                if(window.sosInterval) clearInterval(window.sosInterval);
            }
        }
    </script>
